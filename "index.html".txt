  <p>Enter a website URL. It will automatically add <code>/?scrlcbp=.</code> to the end.</p>

  <input id="urlInput" placeholder="https://example.com">
  <button onclick="loadSite()">Search</button>

  <div id="finalLink"></div>
  <iframe id="browser"></iframe>

  <script>
    function loadSite() {
      let url = document.getElementById("urlInput").value.trim();
      if (!url) {
        alert("Please enter a URL.");
        return;
      }

      // Add protocol if missing
      if (!url.startsWith("http://") && !url.startsWith("https://")) {
        url = "https://" + url;
      }

      let modifiedUrl;

      // Add /?scrlcbp=. correctly
      if (url.includes("?")) {
        modifiedUrl = url + "&scrlcbp=.";
      } else {
        if (!url.endsWith("/")) url += "/";
        modifiedUrl = url + "?scrlcbp=.";
      }

      // Show the modified link
      document.getElementById("finalLink").innerHTML =
        `<strong>Final Link:</strong><br>
         <a href="${modifiedUrl}" target="_blank" rel="noopener noreferrer">
           ${modifiedUrl}
         </a>`;

      // Open in iframe
      document.getElementById("browser").src = modifiedUrl;
    }

    // Optional: Press Enter to search
    document.getElementById("urlInput").addEventListener("keypress", function(e) {
      if (e.key === "Enter") loadSite();
    });
  </script>
